## Solution
```python
import sys

input_data = sys.stdin.read().strip().split('\n')

t = int(input_data[0])
results = []
for i in range(1, t + 1):
    n, k = map(int, input_data[i * 2 - 1].split())
    a = list(map(int, input_data[i * 2].split()))
    count = 0
    for j in range(n):
        for l in range(j + 1, n):
            if a[j] + a[l] == k:
                count += 1
    results.append(str(count))

print('\n'.join(results))
```
