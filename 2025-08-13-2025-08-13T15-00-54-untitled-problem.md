## Solution
```python
import sys

input_data = sys.stdin.read().strip().split('\n')
n = int(input_data[0])
arr = list(map(int, input_data[1].split()))
target_length = int(input_data[2])

# Initialize a dictionary to count occurrences of words
word_count = {}

# Process the list of words
for word in arr:
    if word in word_count:
        word_count[word] += 1
    else:
        word_count[word] = 1

# Collect words with the required length
result = []
for word, count in word_count.items():
    if len(word) == target_length:
        result.append((word, count))

# Output the result in sorted order
result.sort(key=lambda x: x[0])  # Sort by word
for word, count in result:
    print(f'{word} {count}')
```
