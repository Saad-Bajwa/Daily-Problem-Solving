## Solution

```python
import sys

input_data = sys.stdin.read().strip().split('\n')
n = int(input_data[0])
arr = list(map(int, input_data[1].split()))

# Finding the maximum sum of non-adjacent elements
if n == 0:
    print(0)
elif n == 1:
    print(arr[0])
else:
    dp = [0] * n
    dp[0] = arr[0]
    dp[1] = max(arr[0], arr[1])

    for i in range(2, n):
        dp[i] = max(dp[i-1], dp[i-2] + arr[i])

    print(dp[-1])
```
