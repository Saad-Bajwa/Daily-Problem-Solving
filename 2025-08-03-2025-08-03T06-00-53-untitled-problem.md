import sys

input_data = sys.stdin.read().strip().split('\n')
n = int(input_data[0])
arr = list(map(int, input_data[1].split()))

# Initialize the hash set to store the elements seen so far
seen = set()
result = []

for num in arr:
    if -num in seen:
        result.append((-num, num))
    seen.add(num)

# Output the pairs found
for pair in result:
    print(pair[0], pair[1])
