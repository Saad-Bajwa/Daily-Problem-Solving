input_data = sys.stdin.read().strip().split('\n')
n = int(input_data[0])
arr = list(map(int, input_data[1].split()))
target = int(input_data[2])

# Create a set to store seen numbers
d seen = set()

for i in range(n):
    complement = target - arr[i]
    if complement in seen:
        print(f'{arr[i]} {complement}')
        break
    seen.add(arr[i])
