## Solution
```python
import sys

# Read input
input_data = sys.stdin.read().strip().split('\n')
n = int(input_data[0])
arr = list(map(int, input_data[1].split()))

# Function to find the two numbers that add up to zero
found_pair = False
for i in range(n):
    for j in range(i + 1, n):
        if arr[i] + arr[j] == 0:
            print(f'{arr[i]}, {arr[j]}')
            found_pair = True
            break
    if found_pair:
        break
if not found_pair:
    print('No pairs found')
```
