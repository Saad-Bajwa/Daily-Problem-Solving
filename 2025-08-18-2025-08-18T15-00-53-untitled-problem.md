import sys

input_data = sys.stdin.read().strip().split('\n')
n = int(input_data[0])
arr = list(map(int, input_data[1].split()))

# We can use a set to track the elements we've seen so far.
seen = set()
results = []

for num in arr:
    if -num in seen:
        results.append((num, -num))
    seen.add(num)

# Format output to match the expected output format
for pair in results:
    print(f'{pair[0]} {pair[1]}')
