```python
import sys

input_data = sys.stdin.read().strip().split('\n')
n = int(input_data[0])
arr = list(map(int, input_data[1].split()))
target = int(input_data[2])

pairs = []
seen = set()

for number in arr:
    complement = target - number
    if complement in seen:
        pairs.append((complement, number))
    seen.add(number)

for x, y in pairs:
    print(f'{x},{y}')
```
