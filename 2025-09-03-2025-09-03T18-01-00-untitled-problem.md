import sys

input_data = sys.stdin.read().strip().split('\n')

# Read the number of commands
t = int(input_data[0].strip())

# Initialize a variable to store results
results = []

# Iterate through each command
for i in range(1, t + 1):
    command = input_data[i].strip().split('\s+')
    op = command[0]

    if op == 'insert':
        x = int(command[1])
        y = int(command[2])
        # This operation represents adding x at the y-th position of an array
        if y <= len(results):
            results.insert(y, x)
        else:
            results.append(x)

    elif op == 'print':
        # Output the current state of the results list
        print(results)

    elif op == 'remove':
        x = int(command[1])
        # This will remove the first occurrence of x
        results.remove(x)

    elif op == 'append':
        x = int(command[1])
        # Append x to the end of the list
        results.append(x)

    elif op == 'sort':
        # Sort the results list
        results.sort()

    elif op == 'pop':
        # Pop the last element from the results list
        if results:
            results.pop()

    elif op == 'shift':
        # Shift operation is not defined here, using pop to remove first element
        if results:
            results.pop(0)

    elif op == 'reverse':
        # Reverse the order of elements
        results.reverse()
