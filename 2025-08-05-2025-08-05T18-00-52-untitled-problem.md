## Solution
```python
import sys

input_data = sys.stdin.read().strip().split('\n')
n = int(input_data[0])
arr = list(map(int, input_data[1].split()))

max_sum = float('-inf')
max_i = -1
max_j = -1

for i in range(n):
    for j in range(i + 1, n):
        current_sum = arr[i] + arr[j]
        if current_sum > max_sum:
            max_sum = current_sum
            max_i = i
            max_j = j

print(f'{max_i},{max_j}')
```
