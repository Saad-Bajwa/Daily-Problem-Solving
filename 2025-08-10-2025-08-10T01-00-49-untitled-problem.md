## Solution
```python
import sys

input_data = sys.stdin.read().strip().split('\n')
n = int(input_data[0])
arr = list(map(int, input_data[1].split()))

sum_maps = {}
for i in range(n):
    current_sum = arr[i]
    if current_sum in sum_maps:
        sum_maps[current_sum] += 1
    else:
        sum_maps[current_sum] = 1

result = []
for key, count in sum_maps.items():
    if count > 1:
        result.append(f'({key},{count})')

print(' '.join(result))
```
