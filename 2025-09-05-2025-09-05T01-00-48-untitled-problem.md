## Solution

```python
import sys

input_data = sys.stdin.read().strip().split('\n')

n = int(input_data[0])
elements = list(map(int, input_data[1].split()))

# Calculate the maximum sum of any two distinct elements
if n < 2:
    print(0)
else:
    elements.sort(reverse=True)
    max_sum = elements[0] + elements[1]
    print(max_sum)
```
