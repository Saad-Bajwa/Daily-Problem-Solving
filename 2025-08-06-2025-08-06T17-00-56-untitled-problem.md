## Solution
```python
import sys

input_data = sys.stdin.read().strip().split('\n')
n = int(input_data[0])
arr = list(map(int, input_data[1].split()))

max_sum = 0
result = -1

for i in range(n):
    for j in range(i+1, n):
        current_sum = arr[i] + arr[j]
        if current_sum > max_sum:
            max_sum = current_sum
            result = (arr[i], arr[j])

if result != -1:
    print(f'{result[0]} {result[1]}')
else:
    print('No pairs found')
```
