import sys

input_data = sys.stdin.read().strip().split('\n')
n = int(input_data[0])
array = list(map(int, input_data[1].split()))

# Function to find the maximum product of two elements in the array

def max_product(arr):
    max1 = -float('inf')
    max2 = -float('inf')

    for num in arr:
        if num > max1:
            max2 = max1
            max1 = num
        elif num > max2:
            max2 = num

    return max1 * max2

result = max_product(array)
print(result)
