import sys

input_data = sys.stdin.read().strip().split('\n')

# Parse the input
n = int(input_data[0])  # The number of elements
arr = list(map(int, input_data[1].split()))  # The list of elements

def find_max_sum_subarray(k, arr):
    max_sum = float('-inf')
    current_sum = 0
    start = 0

    for end in range(len(arr)):
        current_sum += arr[end]  # Add the next element to the current sum
        # Slide the window if we hit the size 'k'
        if end >= k - 1:
            max_sum = max(max_sum, current_sum)  # Record the max value
            current_sum -= arr[start]  # Remove the element going out of the window
            start += 1  # Move the start to the right

    return max_sum

# The value of k is the third line of input
k = int(input_data[2])

# Call the function to find the maximum sum of the subarray of size k
result = find_max_sum_subarray(k, arr)

# Output the result
print(result)
