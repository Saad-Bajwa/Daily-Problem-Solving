import sys

input_data = sys.stdin.read().strip().split('\n')

# Read number of test cases
T = int(input_data[0])
results = []

for i in range(1, T + 1):
    # Read each case
    n = int(input_data[i])
    # Check if n is even or odd
    if n % 2 == 0:
        results.append('Even')
    else:
        results.append('Odd')

# Print all results
for result in results:
    print(result)
