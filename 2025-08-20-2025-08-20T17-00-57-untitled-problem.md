## Solution
```python
import sys

input_data = sys.stdin.read().strip().split('\n')
n = int(input_data[0])
arr = list(map(int, input_data[1].split()))
set_of_numbers = set(arr)
output = []

for number in arr:
    complement = -number
    if complement in set_of_numbers:
        output.append((number, complement))

# To remove duplicates and print the unique pairs only
unique_pairs = set(tuple(sorted(pair)) for pair in output)
for pair in unique_pairs:
    print(f'({pair[0]}, {pair[1]})')
```
