import sys

input_data = sys.stdin.read().strip().split('\n')
n = int(input_data[0])
arr = list(map(int, input_data[1].split()))
s = 0
results = []

for num in arr:
    s += num
    if s == 0:
        results.append(0)
    if s in results:
        print(f'Found a subarray ending here with sum zero.')
    results.append(s)

if not results:
    print('No subarray with sum zero.')
