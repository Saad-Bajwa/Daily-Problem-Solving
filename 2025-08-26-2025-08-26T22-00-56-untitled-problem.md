import sys

input_data = sys.stdin.read().strip().split('\n')
n = int(input_data[0])
arr = list(map(int, input_data[1].split()))

# Initialize a set to keep track of the unique elements
elements = set()
result = []

for num in arr:
    if num not in elements:
        elements.add(num)
        result.append(num)

print(' '.join(map(str, result)))
