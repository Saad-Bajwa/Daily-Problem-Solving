import sys

input_data = sys.stdin.read().strip().split('\n')

n = int(input_data[0])
a = list(map(int, input_data[1].split()))

# Finding the maximum value in the array
def find_max(arr):
    max_val = arr[0]
    for num in arr:
        if num > max_val:
            max_val = num
    return max_val

max_value = find_max(a)
print(max_value)
