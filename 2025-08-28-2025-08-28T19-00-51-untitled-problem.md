## Solution
```python
import sys

input_data = sys.stdin.read().strip().split('\n')
n = int(input_data[0])
m = int(input_data[1])
arr = list(map(int, input_data[2].split()))

# We need to find subarrays of length m with max sum
max_sum = -float('inf')
current_sum = sum(arr[:m])
max_sum = current_sum

for i in range(m, n):
    current_sum += arr[i] - arr[i - m]
    if current_sum > max_sum:
        max_sum = current_sum

print(max_sum)
```
