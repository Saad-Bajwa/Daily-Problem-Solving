import sys

input_data = sys.stdin.read().strip().split('\n')

n = int(input_data[0])
arr = list(map(int, input_data[1].split()))

# Assuming the problem is to find the maximum sum of any two distinct elements in the array.
if n < 2:
    print(0)
else:
    arr.sort(reverse=True)
    max_sum = arr[0] + arr[1]
    print(max_sum)
