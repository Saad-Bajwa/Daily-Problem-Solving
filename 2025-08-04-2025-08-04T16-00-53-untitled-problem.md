## Solution
```python
import sys

input_data = sys.stdin.read().strip().split('\n')
n = int(input_data[0])
arr = list(map(int, input_data[1].split()))
target = int(input_data[2])

result = False
for i in range(n):
    for j in range(i + 1, n):
        if arr[i] + arr[j] == target:
            print(f'{i},{j}')
            result = True
            break
    if result:
        break
if not result:
    print('No such indices found')
```
