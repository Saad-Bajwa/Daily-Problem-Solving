import sys

input_data = sys.stdin.read().strip().split('\n')
n = int(input_data[0])
arr = list(map(int, input_data[1].split()))

max_value = max(arr)
count_map = {i: 0 for i in range(max_value + 1)}

for num in arr:
    count_map[num] += 1

result = []
for value, count in count_map.items():
    if count > 0:
        result.append(count)

print(' '.join(map(str, result)))
