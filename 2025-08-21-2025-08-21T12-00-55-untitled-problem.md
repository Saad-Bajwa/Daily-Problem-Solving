## Solution
```python
import sys

input_data = sys.stdin.read().strip().split('\n')
n = int(input_data[0])
d = list(map(int, input_data[1].split()))
k = int(input_data[2])

min_diff = float('inf')
result = None

for i in range(n):
    for j in range(i + 1, n):
        diff = abs(d[i] + d[j] - k)
        if diff < min_diff:
            min_diff = diff
            result = (d[i], d[j])

if result:
    print(result[0], result[1])
```
