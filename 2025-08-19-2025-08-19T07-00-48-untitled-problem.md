```python
import sys

input_data = sys.stdin.read().strip().split('\n')
n = int(input_data[0])
arr = list(map(int, input_data[1].split()))

def count_pairs(arr, n):
    count = 0
    for i in range(n):
        for j in range(i + 1, n):
            if arr[i] == arr[j]:
                count += 1
    return count

result = count_pairs(arr, n)
print(result)
```
