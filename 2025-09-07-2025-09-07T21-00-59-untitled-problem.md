import sys

input_data = sys.stdin.read().strip().split('\n')
n = int(input_data[0])
arr = list(map(int, input_data[1].split()))

# Initialize the count of inversions
inversion_count = 0

# Function to merge two halves and count inversions

def merge_and_count(arr, temp_arr, left, mid, right):
    global inversion_count
    i = left    # Starting index for left subarray
    j = mid + 1 # Starting index for right subarray
    k = left    # Starting index to be sorted

    while i <= mid and j <= right:
        if arr[i] <= arr[j]:
            temp_arr[k] = arr[i]
            i += 1
        else:
            temp_arr[k] = arr[j]
            inversion_count += (mid - i + 1) # Counting inversions
            j += 1
        k += 1

    while i <= mid:
        temp_arr[k] = arr[i]
        i += 1
        k += 1

    while j <= right:
        temp_arr[k] = arr[j]
        j += 1
        k += 1

    for i in range(left, right + 1):
        arr[i] = temp_arr[i]

# Function to use the merge sort algorithm and count inversions

def merge_sort_and_count(arr, temp_arr, left, right):
    if left < right:
        mid = (left + right) // 2

        merge_sort_and_count(arr, temp_arr, left, mid)
        merge_sort_and_count(arr, temp_arr, mid + 1, right)

        merge_and_count(arr, temp_arr, left, mid, right)

# Main function to execute the count inversion logic

temp_arr = [0] * n
merge_sort_and_count(arr, temp_arr, 0, n - 1)
print(inversion_count)
