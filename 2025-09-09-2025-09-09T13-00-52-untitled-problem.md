import sys

input_data = sys.stdin.read().strip().split('\n')
n, q = map(int, input_data[0].split())
money = list(map(int, input_data[1].split()))

results = []
for i in range(2, 2 + q):
    x = int(input_data[i])
    # Check if x is in the money list
    if x in money:
        results.append('Yes')
    else:
        results.append('No')

print('\n'.join(results))
