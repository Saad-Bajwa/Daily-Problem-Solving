## Solution
```python
import sys

# Read the entire input and split by new lines
input_data = sys.stdin.read().strip().split('\n')

# First line is the number of integers
n = int(input_data[0])
# Second line is the list of integers, splitting by space and converting to int
arr = list(map(int, input_data[1].split()))
# Last line is the target sum to find
target = int(input_data[2])

# Create a dictionary to store the elements we have seen
seen = {}
result = []

# Iterate through the array
for index, number in enumerate(arr):
    difference = target - number
    # Check if the difference is in the seen dictionary
    if difference in seen:
        result.append((seen[difference], index))  # append the indexes of the two numbers
    # Store the index of the current number in the dictionary
    seen[number] = index

# Print results as required
for i, j in result:
    print(f'{i},{j}')
```
