import sys

input_data = sys.stdin.read().strip().split('\n')
n = int(input_data[0])
arr = list(map(int, input_data[1].split()))

def max_sum_subarray(arr, n):
    max_sum = float('-inf')
    current_sum = 0

    for i in range(n):
        current_sum += arr[i]
        if current_sum > max_sum:
            max_sum = current_sum
        if current_sum < 0:
            current_sum = 0

    return max_sum

result = max_sum_subarray(arr, n)
print(result)
