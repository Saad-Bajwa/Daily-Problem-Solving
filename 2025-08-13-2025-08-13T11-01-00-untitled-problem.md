```python
import sys

input_data = sys.stdin.read().strip().split('\n')

n = int(input_data[0])
m = int(input_data[1])
arr = list(map(int, input_data[2].split()))

# Initialize a set to track seen numbers
seen = set()
for num in arr:
    if (m - num) in seen:
        print('YES')
        break
    seen.add(num)
else:
    print('NO')
```
