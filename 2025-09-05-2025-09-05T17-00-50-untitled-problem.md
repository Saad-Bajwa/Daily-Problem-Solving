import sys

input_data = sys.stdin.read().strip().split('\n')
n = int(input_data[0])
arr = list(map(int, input_data[1].split()))

max_sum = float('-inf')
start = 0
for end in range(len(arr)):
    max_sum = max(max_sum, sum(arr[start:end + 1]))
    while start < end and arr[start] < arr[end]:
        start += 1
        max_sum = max(max_sum, sum(arr[start:end + 1]))

print(max_sum)
