import sys

input_data = sys.stdin.read().strip().split('\n')
n = int(input_data[0])
m = int(input_data[1])
arr = list(map(int, input_data[2].split()))
k = int(input_data[3])

count = 0
for i in range(n):
    for j in range(i + 1, n):
        if arr[i] + arr[j] == k:
            count += 1

print(count)
