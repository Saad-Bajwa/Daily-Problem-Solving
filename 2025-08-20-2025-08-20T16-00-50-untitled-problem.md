```python
import sys

input_data = sys.stdin.read().strip().split('\n')
n = int(input_data[0])
elements = list(map(int, input_data[1].split()))
k = int(input_data[2])

# Function to find pairs that sum to k
count = 0
count_map = {}

for num in elements:
    complement = k - num
    if complement in count_map:
        count += count_map[complement]
    if num in count_map:
        count_map[num] += 1
    else:
        count_map[num] = 1

print(count)
```
