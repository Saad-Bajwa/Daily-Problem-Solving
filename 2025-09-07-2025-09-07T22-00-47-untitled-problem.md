import sys

input_data = sys.stdin.read().strip().split('\n')
n = int(input_data[0])
numbers = list(map(int, input_data[1].split()))

result_sum = sum(numbers)
result_average = result_sum / n

print(f'{result_sum}\n{result_average:.2f}')
