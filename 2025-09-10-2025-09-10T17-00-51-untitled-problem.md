```python
import sys

input_data = sys.stdin.read().strip().split('\n')
n = int(input_data[0])
arr = list(map(int, input_data[1].split()))

max_element = max(arr)
squared_elements = [x**2 for x in arr]
result = [0] * (max_element + 1)

for num in squared_elements:
    if num <= max_element:
        result[num] += 1

for i in range(1, len(result)):
    print(result[i])
```
