import sys

input_data = sys.stdin.read().strip().split('\n')

n = int(input_data[0])
arr = list(map(int, input_data[1].split()))

result = []
max_length = 0
max_lr = None

for start in range(n):
    current_sum = 0
    for end in range(start, n):
        current_sum += arr[end]
        if current_sum == 0:
            length = end - start + 1
            if length > max_length:
                max_length = length
                max_lr = (start, end)

if max_lr:
    print(max_lr[0], max_lr[1])
else:
    print(-1)
