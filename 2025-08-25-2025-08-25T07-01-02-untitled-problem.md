import sys

# Read input data
input_data = sys.stdin.read().strip().split('\n')

# Parse the first line to get the number of elements
n = int(input_data[0])

# Parse the second line to get the list of integers
arr = list(map(int, input_data[1].split()))

# Initialize the count of inversions
inversion_count = 0

def merge_and_count(arr, temp_arr, left, mid, right):
    global inversion_count
    i = left    # Starting index for left subarray
    j = mid + 1 # Starting index for right subarray
    k = left    # Starting index to be sorted

    # Conditions are checked to ensure that i doesn't exceed mid and j doesn't exceed right
    while i <= mid and j <= right:
        if arr[i] <= arr[j]:
            temp_arr[k] = arr[i]
            i += 1
        else:
            # There are mid - i inversions, because all remaining elements in the left subarray
            # (arr[i], arr[i+1], ..., arr[mid]) are greater than arr[j]
            temp_arr[k] = arr[j]
            inversion_count += (mid - i + 1)
            j += 1
        k += 1

    # Copy the remaining elements of left subarray, if any
    while i <= mid:
        temp_arr[k] = arr[i]
        i += 1
        k += 1

    # Copy the remaining elements of right subarray, if any
    while j <= right:
        temp_arr[k] = arr[j]
        j += 1
        k += 1

    # Copy the sorted subarray into Original array
    for i in range(left, right + 1):
        arr[i] = temp_arr[i]


def merge_sort_and_count(arr, temp_arr, left, right):
    if left < right:
        mid = (left + right) // 2

        merge_sort_and_count(arr, temp_arr, left, mid)
        merge_sort_and_count(arr, temp_arr, mid + 1, right)
        merge_and_count(arr, temp_arr, left, mid, right)

# Create a temporary array
temp_arr = [0]*n

# Call the merge_sort_and_count function
merge_sort_and_count(arr, temp_arr, 0, n - 1)

# Print the result
print(inversion_count)
