## Solution
```python
import sys

input_data = sys.stdin.read().strip().split('\n')
n = int(input_data[0])
numbers = list(map(int, input_data[1].split()))
target = int(input_data[2])

result = []
for i in range(n):
    for j in range(i + 1, n):
        if numbers[i] + numbers[j] == target:
            result.append(f'{i},{j}')

if result:
    print('\n'.join(result))
else:
    print('No pairs found')
```
