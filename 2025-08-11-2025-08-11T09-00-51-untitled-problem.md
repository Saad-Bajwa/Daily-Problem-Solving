import sys

input_data = sys.stdin.read().strip().split('\n')
n = int(input_data[0])
arr = list(map(int, input_data[1].split()))

# Using a set to keep track of seen numbers
seen = set()
for number in arr:
    # Check if the number is already in the set
    if number in seen:
        print(number)
        break
    seen.add(number)
