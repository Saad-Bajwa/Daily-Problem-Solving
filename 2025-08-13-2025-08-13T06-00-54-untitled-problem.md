## Solution
```python
import sys

input_data = sys.stdin.read().strip().split('\n')

t = int(input_data[0])
results = []

for i in range(1, t + 1):
    n = int(input_data[i * 2 - 1])
    arr = list(map(int, input_data[i * 2].split()))
    
    result = []
    for j in range(n):
        for k in range(j + 1, n):
            if arr[j] == arr[k]:
                result.append(j)
                result.append(k)
                break

    if result:
        results.append(' '.join(map(str, result)))
    else:
        results.append('-1')

print('\n'.join(results))
```
