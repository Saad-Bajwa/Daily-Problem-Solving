import sys

def main():
    input_data = sys.stdin.read().strip().split('\n')
    n = int(input_data[0])
    arr = list(map(int, input_data[1].split()))

    # Find the maximum sum of any two distinct elements
    max_sum = float('-inf')
    for i in range(n):
        for j in range(i + 1, n):
            max_sum = max(max_sum, arr[i] + arr[j])

    print(max_sum)

if __name__ == '__main__':
    main()
