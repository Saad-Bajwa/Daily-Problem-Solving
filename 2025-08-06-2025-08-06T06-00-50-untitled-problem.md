## Solution

```python
import sys

input_data = sys.stdin.read().strip().split('\n')
n = int(input_data[0])
s = list(map(int, input_data[1].split()))

max_sum = 0
result_pair = (0, 0)
for i in range(n):
    for j in range(i + 1, n):
        current_sum = s[i] + s[j]
        if current_sum > max_sum:
            max_sum = current_sum
            result_pair = (s[i], s[j])

print(result_pair[0], result_pair[1])
```
