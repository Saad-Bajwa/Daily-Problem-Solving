```python
import sys

input_data = sys.stdin.read().strip().split('\n')
n = int(input_data[0])
arr = list(map(int, input_data[1].split()))

# Using a set to track seen numbers
seen = set()
compatible_pairs = []

for number in arr:
    for other in seen:
        if number + other == 0:
            compatible_pairs.append((number, other))
    seen.add(number)

for pair in compatible_pairs:
    print(pair[0], pair[1])
```
