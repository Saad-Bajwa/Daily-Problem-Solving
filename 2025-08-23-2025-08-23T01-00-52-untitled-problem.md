import sys

input_data = sys.stdin.read().strip().split('\n')

t = int(input_data[0])
results = []

for i in range(1, t + 1):
    n = int(input_data[i * 2 - 1])
    arr = list(map(int, input_data[i * 2].split()))
    product = 1
    for num in arr:
        product *= num
    results.append(product)

for result in results:
    print(result)
