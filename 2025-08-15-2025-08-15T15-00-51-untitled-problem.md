```python
import sys

input_data = sys.stdin.read().strip().split('\n')
n = int(input_data[0])
arr = list(map(int, input_data[1].split()))

# Calculate the maximum product of two distinct elements from the array
max_a = -float('inf')
max_b = -float('inf')

for number in arr:
    if number > max_a:
        max_b = max_a
        max_a = number
    elif number > max_b:
        max_b = number

max_product = max_a * max_b
print(max_product)
```
